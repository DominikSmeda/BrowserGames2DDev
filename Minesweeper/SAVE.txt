<script>
const B_EMPTY = 0;
const B_BOMB = 1;
    
    $('#board').click(function()
    {
        alert();
    });
    
    $('#board button').click(function()
    {    
        $('#board button').hide();
        $('#board #load').css('visibility', 'visible');
        setTimeout(function(){ startGame(); }, 500);
    });
    

    
    function startGame()
    {
            //alert("Start!");
            $('#board').html("");
            renderBoard(1);
    }
    
    function renderBoard(level)
    {
        var s_width=0;
        var s_height=0;
        var s_bombs=0;
        
        switch(level) 
        {       
        case 0:
            s_width=15;
            s_height=10;
            s_bombs=5;
            break;
        case 1:
            s_width=18;
            s_height=15;
            s_bombs=12;
            break;
            break;
        default:
            alert("ERROR!");
        }
         
        var board = [];
  
        for(x=0;x<s_width;x++)
        {
            board[x]=[];
            for(y=0;y<s_height;y++)  
            {
               board[x][y]=B_EMPTY;     
            }
        }
        
        for(i=0;i<s_bombs;i++)
        {
            var sx=Math.floor((Math.random() * s_width)+0);
            var sy=Math.floor((Math.random() * s_height)+0);
            board[sx][sy]=B_BOMB; 
            //dopracowac bez powtarzania
        }
        
             //alert(board);
        $('#board').append('<div id="collection"></div>');
        
        for(y=0;y<s_height;y++)
        {
            for(x=0;x<s_width;x++)  
            {
                var id=(y*s_width)+x;
                $('#collection').append('<div id="'+id+'"class="notExpose"></div>');   
            }
            $('#collection').append('<div style="clear:both"></div>');     
        }     
    }

 
    
</script>
